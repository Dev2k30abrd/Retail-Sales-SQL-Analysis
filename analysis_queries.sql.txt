1. Customers per city
SELECT city, COUNT(*) AS total_customers
FROM Customers
GROUP BY city;

2. Total spending per customer (by id)
SELECT customer_id, SUM(total_amount) AS total_spent
FROM Orders
GROUP BY customer_id;

3. Top customers by spending (with names) //INNER JOIN QUERY
SELECT c.name, SUM(o.total_amount) AS total_spent
FROM Customers c
JOIN Orders o
ON c.customer_id = o.customer_id
GROUP BY c.name
ORDER BY total_spent DESC;

4. All customers and their spending (including no orders) //LEFT JOIN QUERY
SELECT c.name, SUM(o.total_amount) AS total_spent
FROM Customers c
LEFT JOIN Orders o
ON c.customer_id = o.customer_id
GROUP BY c.name;

5. Customers who spent more than 30000 //HAVING QUERY
SELECT c.name, SUM(o.total_amount) AS total_spent
FROM Customers c
JOIN Orders o
ON c.customer_id = o.customer_id
GROUP BY c.name
HAVING SUM(o.total_amount) > 30000;

6. Cities with more than 1 customer
SELECT city, COUNT(*) AS total_customers
FROM Customers
GROUP BY city
HAVING COUNT(*) > 1;

   //MULTIPLE JOIN ANALYSIS QUERIES
7. Best selling products (total quantity sold per product)
SELECT p.product_name, SUM(oi.quantity) AS total_sold
FROM Order_Items oi
JOIN Products p
ON oi.product_id = p.product_id
GROUP BY p.product_name
ORDER BY total_sold DESC;

8. Revenue generated by each product category
SELECT p.category, SUM(o.total_amount) AS revenue
FROM Orders o
JOIN Order_Items oi
ON o.order_id = oi.order_id
JOIN Products p
ON oi.product_id = p.product_id
GROUP BY p.category
ORDER BY revenue DESC;

9. Total number of items sold in each order
SELECT order_id, SUM(quantity) AS total_items
FROM Order_Items
GROUP BY order_id;

   //SUBQUERY ANALYSIS QUERIES
10. Customers who spent more than average order value
SELECT c.name, SUM(o.total_amount) AS total_spent
FROM Customers c
JOIN Orders o
ON c.customer_id = o.customer_id
GROUP BY c.name
HAVING SUM(o.total_amount) >
       (SELECT AVG(total_amount) FROM Orders);

11. Products priced above average price
SELECT product_name, price
FROM Products
WHERE price >
      (SELECT AVG(price) FROM Products);

   //DATE ANALYSIS QUERIES
12. Monthly sales trend
SELECT MONTH(order_date) AS month,
       SUM(total_amount) AS monthly_sales
FROM Orders
GROUP BY MONTH(order_date)
ORDER BY month;

13. Orders after a specific date
SELECT *
FROM Orders
WHERE order_date > '2023-05-03';

14. Daily revenue
SELECT order_date, SUM(total_amount) AS daily_revenue
FROM Orders
GROUP BY order_date
ORDER BY order_date;

15. CUSTOMER SEGMENTATION (CASE WHEN)
SELECT c.name,
       SUM(o.total_amount) AS total_spent,
       CASE
           WHEN SUM(o.total_amount) > 50000 THEN 'High Value'
           WHEN SUM(o.total_amount) BETWEEN 20000 AND 50000 THEN 'Medium Value'
           ELSE 'Low Value'
       END AS customer_segment
FROM Customers c
LEFT JOIN Orders o
ON c.customer_id = o.customer_id
GROUP BY c.name
ORDER BY total_spent DESC;
